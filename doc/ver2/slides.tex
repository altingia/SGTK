\section{Цель и задачи}
\begin{frame}[t]{Цель и задачи}
	\begin{block}{Цель}
	Построения скаффолдов по ридам РНК 
	\end{block}
	\begin{block}{Задачи}
	\begin{itemize}
		\item Построение графа связей
		\item Построение скаффолдов основываясь на полученных связях
		\item Создание инструмента для визуализации графа связей между контигами
		\item Сравнение получившихся результатов с результатами других тулов для построения скаффолодов по ридам РНК  
	\end{itemize}
\end{block}
